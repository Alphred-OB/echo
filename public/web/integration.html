Integrate Echo in minutes

Use our simple two-step REST protocol to authenticate user devices using acoustic signatures and non-exportable hardware keys.

Protocol Architecture

The Echo protocol operates as a simple validation pipeline between the client browser, the target authentication server, and the user's registered smartphone.

Step 1: Session Initiation

The laptop web client requests a single-use nonce from your backend. Your backend creates a short-lived session, binds it to the requested username, and issues a 96-bit random nonce to be broadcast acoustically.

// POST /api/login/start
app.post('/api/login/start', (req, res) => {
  const { username } = req.body;
  const sessionId = crypto.randomBytes(16).toString('base64url');
  const nonce = crypto.randomBytes(12).toString('base64url');
  
  db.run(`INSERT INTO login_sessions (id, username, nonce, expires_at) 
          VALUES (?, ?, ?, ?)`, [sessionId, username, nonce, Date.now() + 30000]);
          
  res.json({ sessionId, nonce, ttlMs: 30000 });
});

Step 2: Signature Verification

Once the user's phone decodes the acoustic wave and registers approval, it generates an ECDSA signature over the message pattern `echo-v1|<nonce>|<deviceId>`. The phone POSTs this signature back to your server for validation. You must burn the nonce immediately to reject replay attacks.

// POST /api/login/verify
app.post('/api/login/verify', async (req, res) => {
  const { nonce, deviceId, signature } = req.body;
  const session = db.prepare('SELECT * FROM login_sessions WHERE nonce = ?').get(nonce);
  if (!session || session.used || session.expires_at < Date.now()) {
    return res.status(401).json({ error: 'Invalid or expired session' });
  }

  // Burn the nonce instantly
  db.prepare('UPDATE login_sessions SET used = 1 WHERE nonce = ?').run(nonce);
  
  const device = db.prepare('SELECT pubkey_jwk FROM devices WHERE id = ?').get(deviceId);
  const verified = await verifyECDSASignature(device.pubkey_jwk, `echo-v1|${nonce}|${deviceId}`, signature);
  
  if (verified) {
    db.prepare("UPDATE login_sessions SET status = 'approved' WHERE nonce = ?").run(nonce);
    res.json({ ok: true });
  } else {
    res.status(401).json({ error: 'Signature validation failed' });
  }
});
# POST /api/login/verify
@app.post("/api/login/verify")
def login_verify(payload: VerifyPayload):
    session = db.get_session(payload.nonce)
    if not session or session["used"] or session["expires_at"] < int(time.time() * 1000):
        raise HTTPException(status_code=401, detail="Invalid nonce")
        
    db.burn_nonce(payload.nonce)
    pubkey = db.get_device_key(payload.device_id)
    
    msg = f"echo-v1|{payload.nonce}|{payload.device_id}"
    verified = verify_ecdsa(pubkey, msg, payload.signature)
    
    if verified:
        db.approve_session(payload.nonce)
        return {"ok": True}
    return JSONResponse(status_code=401, content={"error": "Signature failed"})
// POST /api/login/verify
func LoginVerify(w http.ResponseWriter, r *http.Request) {
    var req struct { Nonce, DeviceId, Signature string }
    json.NewDecoder(r.Body).Decode(&req)
    
    session := db.GetSession(req.Nonce)
    if session.Used || session.ExpiresAt < time.Now().UnixMilli() {
        w.WriteHeader(http.StatusUnauthorized)
        return
    }
    
    db.Exec("UPDATE login_sessions SET used = 1 WHERE nonce = ?", req.Nonce)
    pubkey := db.GetDeviceKey(req.DeviceId)
    
    msg := fmt.Sprintf("echo-v1|%s|%s", req.Nonce, req.DeviceId)
    if verifyECDSA(pubkey, msg, req.Signature) {
        db.Exec("UPDATE login_sessions SET status = 'approved' WHERE nonce = ?", req.Nonce)
        json.NewEncoder(w).Encode(map[string]bool{"ok": true})
    } else {
        w.WriteHeader(http.StatusUnauthorized)
    }
}
